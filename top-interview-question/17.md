const AlpMatchedByNum = {
  2: ["a", "b", "c"],
  3: ["d", "e", "f"],
  4: ["g", "h", "i"],
  5: ["j", "k", "l"],
  6: ["m", "n", "o"],
  7: ["p", "q", "r", "s"],
  8: ["t", "u", "v"],
  9: ["w", "x", "y", "z"],
};

const letterComb = (
  didx: number,
  oidx: number,
  max: number,
  res: string,
  array: string[],
  digits: string
) => {
  if (didx >= max) {
    array.push(res);
    return;
  }
  const Alps = AlpMatchedByNum[digits[didx]];
  if (oidx >= Alps.length) {
    return;
  }
  letterComb(didx, oidx + 1, max, res, array, digits);

  res = `${res}${Alps[oidx]}`;
  letterComb(didx + 1, 0, max, res, array, digits);
};

const letterCombinations = (digits: string): string[] => {
  if (digits === "") {
    return [];
  }
  const res = [];
  letterComb(0, 0, digits.length, "", res, digits);
  return res;
};
