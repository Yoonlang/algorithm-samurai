### set 활용

```js
/**
 * @param {number[]} nums
 * @return {boolean}
 */
let containsDuplicate = function (nums) {
  const set = new Set(nums);
  return nums.length !== set.size;
};
```

### object 활용

```js
/**
 * @param {number[]} nums
 * @return {boolean}
 */
let containsDuplicate = function (nums) {
  const numObj = {};

  for (const num of nums) {
    if (!numObj[num]) numObj[num] = true;
    else return true;
  }

  return false;
};
```
