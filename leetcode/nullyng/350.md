/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @return {number[]}
 */
var intersect = function (nums1, nums2) {
  // 11:10~11:20
  const cnt1 = {};
  const cnt2 = {};
  const answer = [];

  for (const num of nums1) {
    if (!cnt1[num]) cnt1[num] = 0;
    cnt1[num]++;
  }
  for (const num of nums2) {
    if (!cnt2[num]) cnt2[num] = 0;
    cnt2[num]++;
  }

  for (let i = 0; i <= 1000; i++) {
    if (cnt1[i] && cnt2[i]) {
      value = Math.min(cnt1[i], cnt2[i]);
      for (let j = 0; j < value; j++) {
        answer.push(i);
      }
    }
  }

  return answer;
};
