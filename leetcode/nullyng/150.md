/**
 * @param {string[]} tokens
 * @return {number}
 */
let evalRPN = function (tokens) {
  const stack = [];
  const operand = { "+": true, "-": true, "*": true, "/": true };

  for (let i = 0; i < tokens.length; i++) {
    if (operand[tokens[i]]) {
      let a = stack.pop();
      let b = stack.pop();

      switch (tokens[i]) {
        case "+":
          stack.push(b + a);
          break;
        case "-":
          stack.push(b - a);
          break;
        case "*":
          stack.push(b * a);
          break;
        case "/":
          stack.push(~~(b / a));
          break;
      }
    } else {
      stack.push(+tokens[i]);
    }
  }

  return stack[0];
};
