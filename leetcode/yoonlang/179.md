/**
 * @param {number[]} nums
 * @return {string}
 */
var largestNumber = function (nums) {
  const res = nums
    .sort((a, b) => {
      return Number(`${b}${a}`) - Number(`${a}${b}`);
    })
    .join("");
  return res[0] === "0" ? "0" : res;
};
