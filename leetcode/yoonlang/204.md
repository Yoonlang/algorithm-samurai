/**
 * @param {number} n
 * @return {number}
 */
var countPrimes = function (n) {
  let res = 0;
  const arr = new Array(n + 1).fill(false);
  for (let i = 2; i < n; i++) {
    if (!arr[i]) {
      res++;
      let t = i;
      while (t < n) {
        arr[t] = true;
        t += i;
      }
    }
  }

  return res;
};
