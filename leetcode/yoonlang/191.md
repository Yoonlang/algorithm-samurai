  let res = 0;
  [...n.toString(2)].forEach((c) => {
    if (c === "1") {

#include <iostream>

class Solution {
public:
    int hammingWeight(uint32_t n) {
        int res = 0;
        bool i = n & 1;
        int t = 31;
        while(t--){
            if(i) res++;
            i = (n >> (31 - t)) & 1;
        }
        if(i) res++;
        return res;
    }
};
