var reverseBits = function (n) {
  let res = 0;
  for (let t = 31; t >= 0; t--) {
    if ((n >> (31 - t)) & 1) {
      res += Math.pow(2, t);
    }
  }
  return res;
};

class Solution {
public:
    uint32_t reverseBits(uint32_t n) {
        uint32_t reverse = 0;

        int t = 32;
        while(t--){
            reverse = reverse << 1;
            if (n & 1) {
                reverse += 1;
            }
            n = n >> 1;
        }

        return reverse;
    }
};
