/**
 * @param {string} s
 * @param {string} t
 * @return {boolean}
 */
var isAnagram = function (s, t) {
  const map = {};

  [...s].forEach((c) => {
    map[c] = map[c] ? map[c] + 1 : 1;
  });

  [...t].forEach((c) => {
    map[c]--;
  });

  return !Object.values(map).some((v) => v !== 0);
};

function isAnagram(s: string, t: string): boolean {
  const o = {};
  [...s].forEach((c) => {
    if (o.hasOwnProperty(c)) {
      o[c]++;
    } else {
      o[c] = 1;
    }
  });

  let flag = false;
  [...t].forEach((c) => {
    if (o.hasOwnProperty(c)) {
      o[c]--;
      if (o[c] === 0) {
        delete o[c];
      }
    } else {
      flag = true;
    }
  });

  if (!flag && Object.keys(o).length === 0) {
    return true;
  } else {
    return false;
  }
}
