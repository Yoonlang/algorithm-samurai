class Solution {
public:
    int numSquares(int n) {
        vector<int> dp(n+1, 9999999);

        for(int i=1; i<=n; i++) {
            int s = floor(sqrt(i));
            while(s > 0) {
                int maxSquareNum = s*s;
                if(i%maxSquareNum) {
                    dp[i] = min(i/maxSquareNum + dp[i%maxSquareNum], dp[i]);
                } else {
                    dp[i] = min(i/maxSquareNum, dp[i]);
                }
                s--;
            }
        }

        return dp[n];
    };
};
