### 비트 연산자를 이용한 풀이

```js
/**
 * @param {number} n
 * @return {number}
 */
let hammingWeight = function (n) {
  let answer = 0;
  while (n > 0) {
    if (n & 1) answer++;
    n >>= 1;
  }
  return answer;
};
```

### toString을 이용한 풀이

```js
/**
 * @param {number} n
 * @return {number}
 */
let hammingWeight = function (n) {
  n = n.toString(2);
  return n.split("").filter((v) => v === "1").length;
};
```
