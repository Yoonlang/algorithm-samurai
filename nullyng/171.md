```js
/**
 * @param {string} columnTitle
 * @return {number}
 */
let titleToNumber = function (columnTitle) {
  let answer = 0;
  let base = 1;

  for (let i = columnTitle.length - 1; i >= 0; i--) {
    answer += base * (columnTitle[i].charCodeAt() - 64);
    base *= 26;
  }

  return answer;
};
```
