/**
 * @param {number[]} nums
 * @param {number} k
 * @return {void} Do not return anything, modify nums in-place instead.
 */
let rotate = function (nums, k) {
  let len = nums.length;
  k = k % len;

  for (let i = 0; i < len - k; i++) {
    nums.push(nums[i]);
  }

  nums.splice(0, len - k);
};
