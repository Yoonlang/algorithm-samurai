/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @param {number[]} nums3
 * @param {number[]} nums4
 * @return {number}
 */
var fourSumCount = function (nums1, nums2, nums3, nums4) {
  let res = 0;
  const map = new Map();

  nums1.forEach((n1) => {
    nums2.forEach((n2) => {
      if (map.has(n1 + n2)) {
        map.set(n1 + n2, map.get(n1 + n2) + 1);
      } else {
        map.set(n1 + n2, 1);
      }
    });
  });

  nums3.forEach((n3) => {
    nums4.forEach((n4) => {
      if (map.has((n3 + n4) * -1)) {
        res += map.get((n3 + n4) * -1);
      }
    });
  });

  return res;
};
